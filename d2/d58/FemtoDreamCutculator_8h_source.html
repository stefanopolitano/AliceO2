91  void checkForSelection(aod::femtodreamparticle::cutContainerType& output, size_t& counter, T1 objectSelection, T2 selectionType)
92  {
94  std::cout << "Selection: " << objectSelection.getSelectionHelper(selectionType) << " - (";
95  auto selVec = objectSelection.getSelections(selectionType);
96  for (auto selIt : selVec) {
97  std::cout << selIt.getSelectionValue() << " ";
98  }
99  std::cout << ")\n > ";
100  std::string in;
101  std::cin >> in;
102  const float input = std::stof(in);
103 
105  bool inputSane = false;
106  for (auto sel : selVec) {
107  if (std::abs(sel.getSelectionValue() - input) < std::abs(1.e-6 * input)) {
108  inputSane = true;
109  }
110  }
111 
113  if (inputSane) {
114  for (auto sel : selVec) {
115  double signOffset;
116  switch (sel.getSelectionType()) {
117  case femtoDreamSelection::SelectionType::kEqual:
118  signOffset = 0.;
119  break;
120  case (femtoDreamSelection::SelectionType::kLowerLimit):
121  case (femtoDreamSelection::SelectionType::kAbsLowerLimit):
122  signOffset = 1.;
123  break;
124  case (femtoDreamSelection::SelectionType::kUpperLimit):
125  case (femtoDreamSelection::SelectionType::kAbsUpperLimit):
126  signOffset = -1.;
127  break;
128  }
129 
131  if (sel.isSelected(input + signOffset * 1.e-6 * input)) {
132  output |= 1UL << counter;
133  }
134  ++counter;
135  }
136  } else {
137  std::cout << "Choice " << in << " not recognized - repeating\n";
138  checkForSelection(output, counter, objectSelection, selectionType);
139  }
140  }
