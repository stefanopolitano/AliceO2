314  trigsel = true;

317

321  zvtxsel = true;

323 
324  bool centmultsel = false;
325  switch (fCentMultEstimator) {
326  case kV0M:

328  centormult = collision.centV0M();
329  centmultsel = true;

331  break;
332  default:

334  }
335  return trigsel and zvtxsel and centmultsel;

415  asone = false;

480  template <typename TrackListObject, typename CollisionIndex>
481  void filterTracks(TrackListObject const& ftracks, CollisionIndex colix)
482  {
483  using namespace filteranalysistask;
484 
485  int acceptedtracks = 0;
486 
487  for (auto& track : ftracks) {
488  /* before track selection */
489  fhPtB->Fill(track.pt());
490  fhEtaB->Fill(track.eta());
491  fhPhiB->Fill(track.phi());
492  fhEtaVsPhiB->Fill(track.phi(), track.eta());
493  fhPtVsEtaB->Fill(track.eta(), track.pt());
494  if (track.sign() > 0) {
495  fhPtPosB->Fill(track.pt());
496  } else {
497  fhPtNegB->Fill(track.pt());
498  }
499 
500  /* track selection */
501  /* tricky because the boolean columns issue */
502  bool asone, astwo;
503  AcceptTrack(track, asone, astwo);
504  if (asone or astwo) {
505  /* the track has been accepted */
506  fhPtA->Fill(track.pt());
507  fhEtaA->Fill(track.eta());
508  fhPhiA->Fill(track.phi());
509  fhEtaVsPhiA->Fill(track.phi(), track.eta());
510  fhPtVsEtaA->Fill(track.eta(), track.pt());
511  if (track.sign() > 0) {
512  fhPtPosA->Fill(track.pt());
513  } else {
514  fhPtNegA->Fill(track.pt());
515  }
516  acceptedtracks++;
517  }
518  scannedtracks(colix, (uint8_t)asone, (uint8_t)astwo, track.pt(), track.eta(), track.phi());
519  }
520  LOGF(info, "Accepted %d reconstructed tracks", acceptedtracks);

590  /* the centrality/multiplicity estimation */
591  if (cfgCentMultEstimator->compare("V0M") == 0) {
592  fCentMultEstimator = kV0M;
593  } else if (cfgCentMultEstimator->compare("NOCM") == 0) {
594  fCentMultEstimator = kNOCM;
595  } else {
596  LOGF(fatal, "Centrality/Multiplicity estimator %s not supported yet", cfgCentMultEstimator->c_str());
597  }
598 
599  /* if the system type is not known at this time, we have to put the initalization somewhere else */
600  fSystem = getSystemType(cfgSystem);
601  fDataType = getGenType(cfgDataType);
602  fPDG = TDatabasePDG::Instance();
603 
604  /* create the output list which will own the task histograms */
605  TList* fOutputList = new TList();
606  fOutputList->SetOwner(true);
607  fOutput.setObject(fOutputList);
608 
609  /* incorporate configuration parameters to the output */
610  fOutputList->Add(new TParameter<Int_t>("TrackType", cfgTrackType, 'f'));
611  fOutputList->Add(new TParameter<Int_t>("TrackOneCharge", trackonecharge, 'f'));
612  fOutputList->Add(new TParameter<Int_t>("TrackTwoCharge", tracktwocharge, 'f'));
613 
614  if ((fDataType == kData) or (fDataType == kMC)) {
615  /* create the reconstructed data histograms */
616  if (fSystem > kPbp) {
617  fhCentMultB = new TH1F("CentralityB", "Centrality before cut; centrality (%)", 100, 0, 100);
618  fhCentMultA = new TH1F("CentralityA", "Centrality; centrality (%)", 100, 0, 100);
619  } else {

1003  double sum2PtPtnw = 0;

1006  double ptavg_1 = 0.0; /* TODO: load ptavg_1 for eta1, phi1 bin */
