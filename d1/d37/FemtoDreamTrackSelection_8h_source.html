190 template <typename T>

192 {
193  switch (pid) {
194  case o2::track::PID::Electron:
195  return track.tpcNSigmaEl();
196  break;
197  case o2::track::PID::Muon:
198  return track.tpcNSigmaMu();
199  break;
200  case o2::track::PID::Pion:
201  return track.tpcNSigmaPi();
202  break;
203  case o2::track::PID::Kaon:
204  return track.tpcNSigmaKa();
205  break;
206  case o2::track::PID::Proton:
207  return track.tpcNSigmaPr();
208  break;
209  case o2::track::PID::Deuteron:
210  return track.tpcNSigmaDe();
211  break;
212  default:
213  return 999.f;
214  break;
215  }

252 template <typename T>
253 bool FemtoDreamTrackSelection::isSelectedMinimal(T const& track)
254 {
255  const auto pT = track.pt();
256  const auto eta = track.eta();
257  const auto tpcNClsF = track.tpcNClsFound();
258  const auto tpcRClsC = track.tpcCrossedRowsOverFindableCls();
259  const auto tpcNClsC = track.tpcNClsCrossedRows();
260  const auto tpcNClsS = track.tpcNClsShared();
261  const auto dcaXY = track.dcaXY();
262  const auto dcaZ = track.dcaZ();
263  const auto dca = std::sqrt(pow(dcaXY, 2.) + pow(dcaZ, 2.));
264  std::vector<float> pidTPC, pidTOF;
265  for (auto it : mPIDspecies) {
266  pidTPC.push_back(getNsigmaTPC(track, it));
267  pidTOF.push_back(getNsigmaTOF(track, it));
268  }
269 
271  const static int nPtMinSel = getNSelections(femtoDreamTrackSelection::kpTMin);
272  const static int nPtMaxSel = getNSelections(femtoDreamTrackSelection::kpTMax);
273  const static int nEtaSel = getNSelections(femtoDreamTrackSelection::kEtaMax);
274  const static int nTPCnMinSel = getNSelections(femtoDreamTrackSelection::kTPCnClsMin);
275  const static int nTPCfMinSel = getNSelections(femtoDreamTrackSelection::kTPCfClsMin);
276  const static int nTPCcMinSel = getNSelections(femtoDreamTrackSelection::kTPCcRowsMin);
277  const static int nTPCsMaxSel = getNSelections(femtoDreamTrackSelection::kTPCsClsMax);
278  const static int nDCAxyMaxSel = getNSelections(femtoDreamTrackSelection::kDCAxyMax);
279  const static int nDCAzMaxSel = getNSelections(femtoDreamTrackSelection::kDCAzMax);
280  const static int nDCAMinSel = getNSelections(femtoDreamTrackSelection::kDCAMin);
281  const static int nPIDnSigmaSel = getNSelections(femtoDreamTrackSelection::kPIDnSigmaMax);
282 
283  const static float pTMin = getMinimalSelection(femtoDreamTrackSelection::kpTMin, femtoDreamSelection::kLowerLimit);
284  const static float pTMax = getMinimalSelection(femtoDreamTrackSelection::kpTMax, femtoDreamSelection::kUpperLimit);
285  const static float etaMax = getMinimalSelection(femtoDreamTrackSelection::kEtaMax, femtoDreamSelection::kAbsUpperLimit);
286  const static float nClsMin = getMinimalSelection(femtoDreamTrackSelection::kTPCnClsMin, femtoDreamSelection::kLowerLimit);
287  const static float fClsMin = getMinimalSelection(femtoDreamTrackSelection::kTPCfClsMin, femtoDreamSelection::kLowerLimit);
288  const static float cTPCMin = getMinimalSelection(femtoDreamTrackSelection::kTPCcRowsMin, femtoDreamSelection::kLowerLimit);
289  const static float sTPCMax = getMinimalSelection(femtoDreamTrackSelection::kTPCsClsMax, femtoDreamSelection::kUpperLimit);
290  const static float dcaXYMax = getMinimalSelection(femtoDreamTrackSelection::kDCAxyMax, femtoDreamSelection::kAbsUpperLimit);
291  const static float dcaZMax = getMinimalSelection(femtoDreamTrackSelection::kDCAzMax, femtoDreamSelection::kAbsUpperLimit);
292  const static float dcaMin = getMinimalSelection(femtoDreamTrackSelection::kDCAMin, femtoDreamSelection::kAbsLowerLimit);
293  const static float nSigmaPIDMax = getMinimalSelection(femtoDreamTrackSelection::kPIDnSigmaMax, femtoDreamSelection::kAbsUpperLimit);
294 
295  if (nPtMinSel > 0 && pT < pTMin) {
296  return false;
297  }
298  if (nPtMaxSel > 0 && pT > pTMax) {
299  return false;
300  }
301  if (nEtaSel > 0 && std::abs(eta) > etaMax) {
302  return false;
303  }
304  if (nTPCnMinSel > 0 && tpcNClsF < nClsMin) {
305  return false;
306  }
307  if (nTPCfMinSel > 0 && tpcRClsC < fClsMin) {
308  return false;
309  }
310  if (nTPCcMinSel > 0 && tpcNClsC < cTPCMin) {
311  return false;
312  }
313  if (nTPCsMaxSel > 0 && tpcNClsS > sTPCMax) {
314  return false;
315  }
316  if (nDCAxyMaxSel > 0 && std::abs(dcaXY) > dcaXYMax) {
317  return false;
318  }
319  if (nDCAzMaxSel > 0 && std::abs(dcaZ) > dcaZMax) {
320  return false;
321  }
322  if (nDCAMinSel > 0 && std::abs(dca) < dcaMin) {
323  return false;
324  }
325  if (nPIDnSigmaSel > 0) {
326  bool isFulfilled = false;
327  for (size_t i = 0; i < pidTPC.size(); ++i) {
328  auto pidTPCVal = pidTPC.at(i);
329  auto pidTOFVal = pidTOF.at(i);
330  auto pidComb = std::sqrt(pidTPCVal * pidTPCVal + pidTOFVal * pidTOFVal);
331  if (std::abs(pidTPCVal) < nSigmaPIDMax || pidComb < nSigmaPIDMax) {
332  isFulfilled = true;
333  }
334  }
335  if (!isFulfilled) {
336  return isFulfilled;
337  }
338  }
339  return true;
340 }

342 template <typename cutContainerType, typename T>
343 std::array<cutContainerType, 2> FemtoDreamTrackSelection::getCutContainer(T const& track)
344 {
345  cutContainerType output = 0;
346  size_t counter = 0;
347  cutContainerType outputPID = 0;
348  size_t counterPID = 0;
349  const auto sign = track.sign();
350  const auto pT = track.pt();
351  const auto eta = track.eta();
352  const auto tpcNClsF = track.tpcNClsFound();
353  const auto tpcRClsC = track.tpcCrossedRowsOverFindableCls();
354  const auto tpcNClsC = track.tpcNClsCrossedRows();
355  const auto tpcNClsS = track.tpcNClsShared();
356  const auto dcaXY = track.dcaXY();
357  const auto dcaZ = track.dcaZ();
358  const auto dca = std::sqrt(pow(dcaXY, 2.) + pow(dcaZ, 2.));
359 
360  std::vector<float> pidTPC, pidTOF;
361  for (auto it : mPIDspecies) {
362  pidTPC.push_back(getNsigmaTPC(track, it));
363  pidTOF.push_back(getNsigmaTOF(track, it));
364  }
365 
366  float observable;
367  for (auto& sel : mSelections) {
368  const auto selVariable = sel.getSelectionVariable();
369  if (selVariable == femtoDreamTrackSelection::kPIDnSigmaMax) {
371  for (size_t i = 0; i < pidTPC.size(); ++i) {
372  auto pidTPCVal = pidTPC.at(i);
373  auto pidTOFVal = pidTOF.at(i);
374  sel.checkSelectionSetBit(pidTPCVal, outputPID, counterPID);
375  auto pidComb = std::sqrt(pidTPCVal * pidTPCVal + pidTOFVal * pidTOFVal);
376  sel.checkSelectionSetBit(pidComb, outputPID, counterPID);
377  }
378 
379  } else {
381  switch (selVariable) {
382  case (femtoDreamTrackSelection::kSign):
383  observable = sign;
384  break;
385  case (femtoDreamTrackSelection::kpTMin):
386  case (femtoDreamTrackSelection::kpTMax):
387  observable = pT;
388  break;
389  case (femtoDreamTrackSelection::kEtaMax):
390  observable = eta;
391  break;
392  case (femtoDreamTrackSelection::kTPCnClsMin):
393  observable = tpcNClsF;
394  break;
395  case (femtoDreamTrackSelection::kTPCfClsMin):
396  observable = tpcRClsC;
397  break;
398  case (femtoDreamTrackSelection::kTPCcRowsMin):
399  observable = tpcNClsC;
400  break;
401  case (femtoDreamTrackSelection::kTPCsClsMax):
402  observable = tpcNClsS;
403  break;
404  case (femtoDreamTrackSelection::kDCAxyMax):
405  observable = dcaXY;
406  break;
407  case (femtoDreamTrackSelection::kDCAzMax):
408  observable = dcaZ;
409  break;
410  case (femtoDreamTrackSelection::kDCAMin):
411  observable = dca;
412  break;
413  case (femtoDreamTrackSelection::kPIDnSigmaMax):
414  break;
415  }
416  sel.checkSelectionSetBit(observable, output, counter);
417  }
418  }
419  return {output, outputPID};
420 }
