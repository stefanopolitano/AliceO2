158  void processSameEvent(o2::aod::FemtoDreamCollision& col,
159  o2::aod::FemtoDreamParticles& parts)
160  {
161 
162  const int multCol = col.multV0M();
163 
165  for (auto& part : partsOne) {
166  if (!isFullPIDSelected(part.pidcut(), part.p(), ConfPIDThreshPartOne, vecTPCPIDPartOne, vecCombPIDPartOne)) {
167  continue;
168  }
169  trackHistoPartOne.fillQA(part);
170  }
171 
172  if (!ConfIsSame) {
173  for (auto& part : partsTwo) {
174  if (!isFullPIDSelected(part.pidcut(), part.p(), ConfPIDThreshPartTwo, vecTPCPIDPartTwo, vecCombPIDPartTwo)) {
175  continue;
176  }
177  trackHistoPartTwo.fillQA(part);
178  }
179  }
180 
182  for (auto& [p1, p2] : combinations(partsOne, partsTwo)) {
183  if (!isFullPIDSelected(p1.pidcut(), p1.p(), ConfPIDThreshPartOne, vecTPCPIDPartOne, vecCombPIDPartOne) || !isFullPIDSelected(p2.pidcut(), p2.p(), ConfPIDThreshPartTwo, vecTPCPIDPartTwo, vecCombPIDPartTwo)) {
184  continue;
185  }
186 
187  // track cleaning
188  if (!pairCleaner.isCleanPair(p1, p2, parts)) {
189  continue;
190  }
191 
192  sameEventCont.setPair(p1, p2, multCol);
193  }
194  }
