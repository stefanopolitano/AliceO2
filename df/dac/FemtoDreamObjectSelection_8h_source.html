75  void setSelection(std::vector<FemtoDreamSelection<selValDataType, selVariable>>& sels)
76  {
78  switch (sels.at(0).getSelectionType()) {
79  case (femtoDreamSelection::SelectionType::kUpperLimit):
80  case (femtoDreamSelection::SelectionType::kAbsUpperLimit):
81  std::sort(sels.begin(), sels.end(), [](FemtoDreamSelection<selValDataType, selVariable> a, FemtoDreamSelection<selValDataType, selVariable> b) {
82  return a.getSelectionValue() > b.getSelectionValue();
83  });
84  break;
85  case (femtoDreamSelection::SelectionType::kLowerLimit):
86  case (femtoDreamSelection::SelectionType::kAbsLowerLimit):
87  case (femtoDreamSelection::SelectionType::kEqual):
88  std::sort(sels.begin(), sels.end(), [](FemtoDreamSelection<selValDataType, selVariable> a, FemtoDreamSelection<selValDataType, selVariable> b) {

90  });
91  break;

104  selValDataType getMinimalSelection(selVariable selVar, femtoDreamSelection::SelectionType selType)
105  {
106  selValDataType minimalSel;
107  switch (selType) {
108  case (femtoDreamSelection::SelectionType::kUpperLimit):
109  case (femtoDreamSelection::SelectionType::kAbsUpperLimit):
110  minimalSel = -999.e9;
111  break;
112  case (femtoDreamSelection::SelectionType::kLowerLimit):
113  case (femtoDreamSelection::SelectionType::kAbsLowerLimit):
114  case (femtoDreamSelection::SelectionType::kEqual):
115  minimalSel = 999.e9;
116  break;
117  }
118 
119  for (auto sel : mSelections) {
120  if (sel.getSelectionVariable() == selVar) {
121  switch (sel.getSelectionType()) {
122  case (femtoDreamSelection::SelectionType::kUpperLimit):
123  case (femtoDreamSelection::SelectionType::kAbsUpperLimit):
124  if (minimalSel < sel.getSelectionValue()) {
125  minimalSel = sel.getSelectionValue();
126  }
127  break;
128  case (femtoDreamSelection::SelectionType::kLowerLimit):
129  case (femtoDreamSelection::SelectionType::kAbsLowerLimit):
130  case (femtoDreamSelection::SelectionType::kEqual):
131  if (minimalSel > sel.getSelectionValue()) {
132  minimalSel = sel.getSelectionValue();
133  }
134  break;
135  }
136  }
137  }
138  return minimalSel;
139  }
