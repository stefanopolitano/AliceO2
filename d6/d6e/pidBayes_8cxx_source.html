105  template <ProbType detIndex, o2::track::PID::ID pid>
106  bool checkEnabled()
107  {
108  static_assert(detIndex < kNDet && detIndex >= 0);
109  if (!enabledDet[detIndex]) {
110  LOG(DEBUG) << "Detector " << detectorName[detIndex] << " disabled";
111  return false; // Setting the probability to 1 if the detector is disabled
112  }
113  LOG(DEBUG) << "Detector " << detectorName[detIndex] << " enabled with " << enabledSpecies.size() << " species";
114 
115  for (const auto enabledPid : enabledSpecies) { // Checking that the species is enabled
116  LOG(DEBUG) << "Testing " << PID::getName(enabledPid) << " " << (int)enabledPid << " vs " << (int)pid;
117  if (enabledPid == pid) {
118  LOG(DEBUG) << "Particle " << PID::getName(enabledPid) << " enabled";
119  Probability[detIndex][pid] = 1.f / enabledSpecies.size(); // set flat distribution (no decision yet)
120  return true;
121  }
122  }
123  return false;
124  }

408  void ComputeBayesProbabilities()
409  {
410  LOG(DEBUG) << "Computing bayes probabilities";
411 
412  float sum = 0.;
413  for (const auto enabledPid : enabledSpecies) {
414  LOG(DEBUG) << "Adding " << Probability[kMerged][enabledPid] << " with prior " << Probability[kPrior][enabledPid];
415  sum += Probability[kMerged][enabledPid] * Probability[kPrior][enabledPid];
416  }
417  if (sum <= 0) {
418  LOG(WARNING) << "Invalid probability densities or prior probabilities";
419  for (int i = 0; i < Probability[kBayesian].size(); i++) {
420  Probability[kBayesian][i] = 1.f / Probability[kBayesian].size();
421  }
422  return;
423  }
424  for (const auto enabledPid : enabledSpecies) {
425  Probability[kBayesian][enabledPid] = Probability[kMerged][enabledPid] * Probability[kPrior][enabledPid] / sum;
426  // if (probDensityMism) {
427  // probDensityMism[enabledPid] *= Probability[kPrior][enabledPid] / sum;
428  // }
429  LOG(DEBUG) << "For " << PID::getName(enabledPid) << " merged prob. " << Probability[kMerged][enabledPid] << " prior " << Probability[kPrior][enabledPid] << " sum " << sum << " bayesian Probability: " << Probability[kBayesian][enabledPid];
430  }

531  template <typename T>
532  void makelogaxis(T h)
533  {
534  if (logAxis == 0) {
535  return;
536  }
537  const int nbins = h->GetNbinsX();
538  double binp[nbins + 1];
539  double max = h->GetXaxis()->GetBinUpEdge(nbins);
540  double min = h->GetXaxis()->GetBinLowEdge(1);
541  if (min <= 0) {
542  min = 0.00001;
543  }
544  double lmin = TMath::Log10(min);
545  double ldelta = (TMath::Log10(max) - lmin) / ((double)nbins);
546  for (int i = 0; i < nbins; i++) {
547  binp[i] = exp(TMath::Log(10) * (lmin + i * ldelta));
548  }
549  binp[nbins] = max + 1;
550  h->GetXaxis()->Set(nbins, binp);
551  }
