80  mHistogramRegistry->add((folderName + "relPairkT").c_str(), "; #it{k}_{T} (GeV/#it{c}); Entries", kTH1F, {kTAxis});
81  mHistogramRegistry->add((folderName + "relPairkstarkT").c_str(), ("; " + femtoObs + "; #it{k}_{T} (GeV/#it{c})").c_str(), kTH2F, {femtoObsAxis, kTAxis});
82  mHistogramRegistry->add((folderName + "relPairkstarmT").c_str(), ("; " + femtoObs + "; #it{m}_{T} (GeV/#it{c}^{2})").c_str(), kTH2F, {femtoObsAxis, mTAxis});

100  template <typename T>
101  void setPair(T const& part1, T const& part2, const int mult)
102  {
103  float femtoObs;
104  if constexpr (mFemtoObs == femtoDreamContainer::Observable::kstar) {
105  femtoObs = FemtoDreamMath::getkstar(part1, mMassOne, part2, mMassTwo);
106  }
107  const float kT = FemtoDreamMath::getkT(part1, mMassOne, part2, mMassTwo);
108  const float mT = FemtoDreamMath::getmT(part1, mMassOne, part2, mMassTwo);
109 
110  if (mHistogramRegistry) {
111  mHistogramRegistry->fill(HIST(mFolderSuffix[mEventType]) + HIST("relPairDist"), femtoObs);
112  mHistogramRegistry->fill(HIST(mFolderSuffix[mEventType]) + HIST("relPairkT"), kT);
113  mHistogramRegistry->fill(HIST(mFolderSuffix[mEventType]) + HIST("relPairkstarkT"), femtoObs, kT);
114  mHistogramRegistry->fill(HIST(mFolderSuffix[mEventType]) + HIST("relPairkstarmT"), femtoObs, mT);
115  mHistogramRegistry->fill(HIST(mFolderSuffix[mEventType]) + HIST("relPairkstarMult"), femtoObs, mult);
116  }
117  }
118 
119  protected:
