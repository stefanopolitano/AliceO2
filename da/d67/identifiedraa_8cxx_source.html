114  template <std::size_t i, typename T1, typename T2>
115  void fillHistograms_MC(T1 const& tracks, T2 const& mcParticles)
116  {
117  for (auto& track : tracks) {
118  const auto mcParticle = track.mcParticle();
119  for (int j = 0; j < 16; j++) {
120  if (globalTrackswoPrim.IsSelected(track, static_cast<TrackSelection::TrackCuts>(j))) {
121  histos.fill(HIST("TrackCut"), j);
122  }
123  }
124  if (mcParticle.pdgCode() * pdg_num[i] <= 0) {
125  continue;
126  }
127  if (std::abs(mcParticle.eta()) > 0.8) {
128  continue;
129  }
130  float mass;
131  if ((i == 0) || (i == 1)) {
132  mass = 0.13957; // pion mass
133  }
134  if ((i == 2) || (i == 3)) {
135  mass = 0.49367; // kaon mass
136  }
137  if ((i == 4) || (i == 5)) {
138  mass = 0.93827; // proton mass
139  }
140  const float energy = std::sqrt(mass * mass + track.p() * track.p());
141  if (std::abs(0.5f * std::log((energy + track.pz()) / (energy - track.pz()))) > 0.5) {
142  continue;
143  }
144  if (!globalTrackswoPrim.IsSelected(track)) {
145  continue;
146  }
147  if (!MC::isPhysicalPrimary(mcParticle)) {
148  if (mcParticle.pdgCode() == pdg_num[i]) {
149  histos.fill(HIST(dca_xy_sec[i]), track.pt(), track.dcaXY());
150  histos.fill(HIST(dca_z_sec[i]), track.pt(), track.dcaZ());
151  }
152  continue;
153  }
154  if (!globalTracks.IsSelected(track)) {
155  continue;
156  }
157  if (mcParticle.pdgCode() != pdg_num[i]) {
158  if (track.hasTOF()) {
159  if constexpr ((i == 0) || (i == 1)) {
160  histos.fill(HIST(tof_MC_full[i]), track.pt(), track.tofNSigmaPi());
161  } else if constexpr ((i == 2) || (i == 3)) {
162  histos.fill(HIST(tof_MC_full[i]), track.pt(), track.tofNSigmaKa());
163  } else if constexpr ((i == 4) || (i == 5)) {
164  histos.fill(HIST(tof_MC_full[i]), track.pt(), track.tofNSigmaPr());
165  }
166  }
167  continue;
168  }
169  histos.fill(HIST(dca_xy_prm[i]), track.pt(), track.dcaXY());
170  histos.fill(HIST(dca_z_prm[i]), track.pt(), track.dcaZ());
171  if (track.hasTOF()) {
172  histos.fill(HIST(matching_num[i]), mcParticle.pt());
173  }
174  histos.fill(HIST(pt_num[i]), mcParticle.pt());
175 
176  if (track.hasTOF()) {
177  if constexpr ((i == 0) || (i == 1)) {
178  histos.fill(HIST(tof_MC[i]), track.pt(), track.tofNSigmaPi());
179  } else if constexpr ((i == 2) || (i == 3)) {
180  histos.fill(HIST(tof_MC[i]), track.pt(), track.tofNSigmaKa());
181  } else if constexpr ((i == 4) || (i == 5)) {
182  histos.fill(HIST(tof_MC[i]), track.pt(), track.tofNSigmaPr());
183  }
184  }
185  }
186  for (auto& particle : mcParticles) {
187  //if (std::abs(particle.eta()) > 0.8) {
188  // continue;
189  // }
190  if (std::abs(0.5f * std::log((particle.e() + particle.pz()) / (particle.e() - particle.pz()))) > 0.5) {
191  continue;
192  }
193  if (!MC::isPhysicalPrimary(particle)) {
194  continue;
195  }
196  if (particle.pdgCode() != pdg_num[i]) {
197  continue;
198  }
199 
200  histos.fill(HIST(pt_den[i]), particle.pt());
201  }
202  }

221  template <std::size_t i, typename T>
222  void fillHistogramsData(T const& tracks)
223  {
224  for (auto& track : tracks) {
225  if (std::abs(track.eta()) > 0.8) {
226  continue;
227  }
228  if constexpr ((i == 0) || (i == 2) || (i == 4)) {
229  if (track.sign() < 0) {
230  continue;
231  }
232  } else if constexpr ((i == 1) || (i == 3) || (i == 5)) {
233  if (track.sign() > 0) {
234  continue;
235  }
236  }
237  float mass;
238  if constexpr ((i == 0) || (i == 1)) {
239  mass = 0.13957; // pion mass
240  } else if constexpr ((i == 2) || (i == 3)) {
241  mass = 0.49367; // kaon mass
242  } else if constexpr ((i == 4) || (i == 5)) {
243  mass = 0.93827; // proton mass
244  }
245  const float energy = std::sqrt(mass * mass + track.p() * track.p());
246  if (std::abs(0.5f * std::log((energy + track.pz()) / (energy - track.pz()))) > 0.5) {
247  continue;
248  }
249 
250  if (!globalTrackswoPrim.IsSelected(track)) {
251  histos.fill(HIST(wo_track_sel[i]), track.pt());
252  continue;
253  }
254  histos.fill(HIST(w_track_sel[i]), track.pt());
255  float nsigmaTOF = -999.f;
256  float nsigmaTPC = -999.f;
257  if constexpr ((i == 0) || (i == 1)) {
258  nsigmaTOF = track.tofNSigmaPi();
259  nsigmaTPC = track.tpcNSigmaPi();
260  } else if constexpr ((i == 2) || (i == 3)) {
261  nsigmaTOF = track.tofNSigmaKa();
262  nsigmaTPC = track.tpcNSigmaKa();
263  } else if constexpr ((i == 4) || (i == 5)) {
264  nsigmaTOF = track.tofNSigmaPr();
265  nsigmaTPC = track.tpcNSigmaPr();
266  }
267  if (std::abs(nsigmaTOF) < 2.f) {
268  histos.fill(HIST(dca_xy[i]), track.pt(), track.dcaXY());
269  histos.fill(HIST(dca_z[i]), track.pt(), track.dcaZ());
270  }
271  if (!globalTracks.IsSelected(track)) {
272  continue;
273  }
274  if (track.hasTOF()) {
275  histos.fill(HIST(tof[i]), track.pt(), nsigmaTOF);
276  }
277  histos.fill(HIST(tpc[i]), track.pt(), nsigmaTPC);
278  }
279  }

302  return;

304  histos.fill(HIST(num_events), 2);
305  fillHistogramsData<0>(tracks);
306  fillHistogramsData<1>(tracks);
307  fillHistogramsData<2>(tracks);
308  fillHistogramsData<3>(tracks);
