42  {
43  std::vector<double> ptBinning = {0.5, 0.6, 0.7, 0.8, 0.9, 1.0, 1.1, 1.2, 1.3, 1.4, 1.5, 1.6,
44  1.8, 2.0, 2.2, 2.4, 2.8, 3.2, 3.6, 4., 5., 6., 8., 10., 12., 14.};
45  std::vector<double> centBinning = {0., 1., 5., 10., 20., 30., 40., 50., 70., 100.};

49 
50  spectra.add("histRecVtxZData", "collision z position", HistType::kTH1F, {{200, -20., +20., "z position (cm)"}});
51  spectra.add("histKeepEventData", "skimming histogram", HistType::kTH1F, {{2, -0.5, +1.5, "true: keep event, false: reject event"}});
52  spectra.add("histTpcSignalData", "Specific energy loss", HistType::kTH2F, {{600, -6., 6., "#it{p} (GeV/#it{c})"}, {1400, 0, 1400, "d#it{E} / d#it{X} (a. u.)"}});
53  spectra.add("histTofSignalData", "TOF signal", HistType::kTH2F, {{600, -6., 6., "#it{p} (GeV/#it{c})"}, {500, 0.0, 1.0, "#beta (TOF)"}});
54  spectra.add("histTpcNsigmaData", "n-sigma TPC", HistType::kTH2F, {ptAxis, {200, -100., +100., "n#sigma_{He} (a. u.)"}});
55  spectra.add("histDcaVsPtData", "dca vs Pt", HistType::kTH2F, {ptAxis, {400, -0.2, 0.2, "dca"}});
56  spectra.add("histInvMassData", "Invariant mass", HistType::kTH1F, {{600, 5.0, +15., "inv. mass GeV/c^{2}"}});

72  void process(soa::Filtered<soa::Join<aod::Collisions, aod::EvSels>>::iterator const& collision, TrackCandidates const& tracks)
73  {
74  //
75  // collision process loop
76  //
77  bool keepEvent = kFALSE;
78  //
79  spectra.fill(HIST("histRecVtxZData"), collision.posZ());
80  //
81  std::vector<TLorentzVector> posTracks;
82  std::vector<TLorentzVector> negTracks;
83  //
84  for (auto track : tracks) { // start loop over tracks
85 
86  TLorentzVector lorentzVector{};
87  lorentzVector.SetPtEtaPhiM(track.pt() * 2.0, track.eta(), track.phi(), constants::physics::MassHelium3);
88  if (lorentzVector.Rapidity() < yMin || lorentzVector.Rapidity() > yMax) {
89  continue;
90  }
91  //
92  // fill QA histograms
93  //
94  float nSigmaHe3 = track.tpcNSigmaHe();
95  nSigmaHe3 += 94.222101 * TMath::Exp(-0.905203 * track.tpcInnerParam());
96  //
97  spectra.fill(HIST("histTpcSignalData"), track.tpcInnerParam() * track.sign(), track.tpcSignal());
98  if (track.sign() < 0) {
99  spectra.fill(HIST("histTpcNsigmaData"), track.pt() * 2.0, nSigmaHe3);
100  }
101  //
102  // check offline-trigger (skimming) condidition
103  //
104  if (nSigmaHe3 > nsigmacutLow && nSigmaHe3 < nsigmacutHigh) {
105  keepEvent = kTRUE;
106  if (track.sign() < 0) {
107  spectra.fill(HIST("histDcaVsPtData"), track.pt() * 2.0, track.dcaXY());
108  }
109  //
110  // store tracks for invariant mass calculation
111  //
112  if (track.sign() < 0) {
113  negTracks.push_back(lorentzVector);
114  }
115  if (track.sign() > 0) {
116  posTracks.push_back(lorentzVector);
117  }
118  //
119  // calculate beta
120  //
121  if (!track.hasTOF()) {
122  continue;
123  }
124  Float_t tofTime = track.tofSignal();
125  Float_t tofLength = track.length();
126  Float_t beta = tofLength / (TMath::C() * 1e-10 * tofTime);
127  spectra.fill(HIST("histTofSignalData"), track.tpcInnerParam() * track.sign(), beta);
128  }
129 
130  } // end loop over tracks
131  //
132  // fill trigger (skimming) results
133  //
134  spectra.fill(HIST("histKeepEventData"), keepEvent);
135  //
136  // calculate invariant mass
137  //
138  for (Int_t iPos = 0; iPos < posTracks.size(); iPos++) {
139  TLorentzVector& vecPos = posTracks[iPos];
140  for (Int_t jNeg = 0; jNeg < negTracks.size(); jNeg++) {
141  TLorentzVector& vecNeg = negTracks[jNeg];
142  TLorentzVector vecMother = vecPos + vecNeg;
143  spectra.fill(HIST("histInvMassData"), vecMother.M());
144  }
145  }
146  }
147 };
