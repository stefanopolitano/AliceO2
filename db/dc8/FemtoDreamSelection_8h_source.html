74  bool isSelected(selValDataType observable)
75  {
76  switch (mSelType) {
77  case (femtoDreamSelection::SelectionType::kUpperLimit):
78  return (observable < mSelVal);
79  case (femtoDreamSelection::SelectionType::kAbsUpperLimit):
80  return (std::abs(observable) < mSelVal);
81  break;
82  case (femtoDreamSelection::SelectionType::kLowerLimit):
83  return (observable > mSelVal);
84  case (femtoDreamSelection::SelectionType::kAbsLowerLimit):
85  return (std::abs(observable) > mSelVal);
86  break;
87  case (femtoDreamSelection::SelectionType::kEqual):
89  return (std::abs(observable - mSelVal) < std::abs(mSelVal * 1e-6));
90  break;
91  }
92  return false;
93  }
